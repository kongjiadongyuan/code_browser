19#include <sys/cdefs.h> /* Needs to come before <hesiod.h>. */
20#include <hesiod.h>
21#include <resolv.h>
22#include <stddef.h>
23
24#include "nss_hesiod.h"

26void *
27_nss_hesiod_init (void)
28{
29 void *context;
30
31 if (hesiod_init (&context) == -1)
32 return NULL;
33
34 /* Use the default (per-thread) resolver state. */
35 __hesiod_res_set (context, &_res, NULL);
36
37 return context;
38}
