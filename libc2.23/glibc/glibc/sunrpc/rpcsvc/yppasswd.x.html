33/*
34 * YP password update protocol
35 * Requires unix authentication
36 */
37program YPPASSWDPROG {
38 version YPPASSWDVERS {
39 /*
40 * Update my passwd entry
41 */
42 int
43 YPPASSWDPROC_UPDATE(yppasswd) = 1;
44 } = 1;
45} = 100009;
46
47
48struct passwd {
49 string pw_name<>; /* username */
50 string pw_passwd<>; /* encrypted password */
51 int pw_uid; /* user id */
52 int pw_gid; /* group id */
53 string pw_gecos<>; /* in real life name */
54 string pw_dir<>; /* home directory */
55 string pw_shell<>; /* default shell */
56};
57
58struct yppasswd {
59 string oldpass<>; /* unencrypted old password */
60 passwd newpw; /* new passwd entry */
61};
