31struct task_security_struct {
32 u32 osid; /* SID prior to last execve */
33 u32 sid; /* current SID */
34 u32 exec_sid; /* exec SID */
35 u32 create_sid; /* fscreate SID */
36 u32 keycreate_sid; /* keycreate SID */
37 u32 sockcreate_sid; /* fscreate SID */
38};
39
40/*
41 * get the subjective security ID of the current task
42 */
43static inline u32 current_sid(void)
44{
45 const struct task_security_struct *tsec = current_security();
46
47 return tsec->sid;
48}

56struct inode_security_struct {
57 struct inode *inode; /* back pointer to inode object */
58 struct list_head list; /* list of inode_security_struct */
59 u32 task_sid; /* SID of creating task */
60 u32 sid; /* SID of this object */
61 u16 sclass; /* security class of this object */
62 unsigned char initialized; /* initialization flag */
63 spinlock_t lock;
64};
65
66struct file_security_struct {
67 u32 sid; /* SID of open file description */
68 u32 fown_sid; /* SID of file owner (for SIGIO) */
69 u32 isid; /* SID of inode at the time of file open */
70 u32 pseqno; /* Policy seqno at the time of file open */
71};

73struct superblock_security_struct {
74 struct super_block *sb; /* back pointer to sb object */
75 u32 sid; /* SID of file system superblock */
76 u32 def_sid; /* default SID for labeling */
77 u32 mntpoint_sid; /* SECURITY_FS_USE_MNTPOINT context for files */
78 unsigned short behavior; /* labeling behavior */
79 unsigned short flags; /* which mount options were specified */
80 struct mutex lock;
81 struct list_head isec_head;
82 spinlock_t isec_lock;
83};

115struct sk_security_struct {

157extern struct lsm_blob_sizes selinux_blob_sizes;
158static inline struct task_security_struct *selinux_cred(const struct cred *cred)
159{
160 return cred->security + selinux_blob_sizes.lbs_cred;
161}
162
163static inline struct file_security_struct *selinux_file(const struct file *file)
164{
165 return file->f_security + selinux_blob_sizes.lbs_file;
166}
167
168static inline struct inode_security_struct *selinux_inode(
169 const struct inode *inode)
170{
171 if (unlikely(!inode->i_security))
172 return NULL;
173 return inode->i_security + selinux_blob_sizes.lbs_inode;
174}
175
176static inline struct msg_security_struct *selinux_msg_msg(
177 const struct msg_msg *msg_msg)
178{
179 return msg_msg->security + selinux_blob_sizes.lbs_msg_msg;
180}
181
182static inline struct ipc_security_struct *selinux_ipc(
183 const struct kern_ipc_perm *ipc)
184{
185 return ipc->security + selinux_blob_sizes.lbs_ipc;
186}
