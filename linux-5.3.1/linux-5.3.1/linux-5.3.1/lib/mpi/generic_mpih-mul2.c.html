17#include "longlong.h"
18
19mpi_limb_t
20mpihelp_addmul_1(mpi_ptr_t res_ptr, mpi_ptr_t s1_ptr,
21 mpi_size_t s1_size, mpi_limb_t s2_limb)
22{
23 mpi_limb_t cy_limb;
24 mpi_size_t j;
25 mpi_limb_t prod_high, prod_low;
26 mpi_limb_t x;
27
28 /* The loop counter and index J goes from -SIZE to -1. This way
29 * the loop becomes faster. */
30 j = -s1_size;
31 res_ptr -= j;
32 s1_ptr -= j;
33
34 cy_limb = 0;
35 do {
36 umul_ppmm(prod_high, prod_low, s1_ptr[j], s2_limb);
37
38 prod_low += cy_limb;
39 cy_limb = (prod_low < cy_limb ? 1 : 0) + prod_high;
40
41 x = res_ptr[j];
42 prod_low = x + prod_low;
43 cy_limb += prod_low < x ? 1 : 0;
44 res_ptr[j] = prod_low;
45 } while (++j);
46 return cy_limb;
47}
