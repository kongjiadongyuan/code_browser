18#define CMAC_TLEN 8 /* CMAC TLen = 64 bits (8 octets) */
19#define CMAC_TLEN_256 16 /* CMAC TLen = 128 bits (16 octets) */
20#define AAD_LEN 20
21
22static const u8 zero[CMAC_TLEN_256];
23
24void ieee80211_aes_cmac(struct crypto_shash *tfm, const u8 *aad,
25 const u8 *data, size_t data_len, u8 *mic)
26{
27 SHASH_DESC_ON_STACK(desc, tfm);
28 u8 out[AES_BLOCK_SIZE];
29
30 desc->tfm = tfm;
31
32 crypto_shash_init(desc);
33 crypto_shash_update(desc, aad, AAD_LEN);
34 crypto_shash_update(desc, data, data_len - CMAC_TLEN);
35 crypto_shash_finup(desc, zero, CMAC_TLEN, out);
36
37 memcpy(mic, out, CMAC_TLEN);
38}
39
40void ieee80211_aes_cmac_256(struct crypto_shash *tfm, const u8 *aad,
41 const u8 *data, size_t data_len, u8 *mic)
42{
43 SHASH_DESC_ON_STACK(desc, tfm);
44
45 desc->tfm = tfm;
46
47 crypto_shash_init(desc);
48 crypto_shash_update(desc, aad, AAD_LEN);
49 crypto_shash_update(desc, data, data_len - CMAC_TLEN_256);
50 crypto_shash_finup(desc, zero, CMAC_TLEN_256, mic);
51}
52
53struct crypto_shash *ieee80211_aes_cmac_key_setup(const u8 key[],
54 size_t key_len)
55{
56 struct crypto_shash *tfm;
57
58 tfm = crypto_alloc_shash("cmac(aes)", 0, 0);
59 if (!IS_ERR(tfm))
60 crypto_shash_setkey(tfm, key, key_len);
61
62 return tfm;
63}
64
65void ieee80211_aes_cmac_key_free(struct crypto_shash *tfm)
66{
67 crypto_free_shash(tfm);
68}
