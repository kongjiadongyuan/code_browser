6#ifndef __RC_MINSTREL_HT_H
7#define __RC_MINSTREL_HT_H
8
9/*
10 * The number of streams can be changed to 2 to reduce code
11 * size and memory footprint.
12 */
13#define MINSTREL_MAX_STREAMS 4
14#define MINSTREL_HT_STREAM_GROUPS 4 /* BW(=2) * SGI(=2) */
15#define MINSTREL_VHT_STREAM_GROUPS 6 /* BW(=3) * SGI(=2) */
16
17#define MINSTREL_HT_GROUPS_NB (MINSTREL_MAX_STREAMS * \
18 MINSTREL_HT_STREAM_GROUPS)
19#define MINSTREL_VHT_GROUPS_NB (MINSTREL_MAX_STREAMS * \
20 MINSTREL_VHT_STREAM_GROUPS)
21#define MINSTREL_CCK_GROUPS_NB 1
22#define MINSTREL_GROUPS_NB (MINSTREL_HT_GROUPS_NB + \
23 MINSTREL_VHT_GROUPS_NB + \
24 MINSTREL_CCK_GROUPS_NB)
25
26#define MINSTREL_HT_GROUP_0 0
27#define MINSTREL_CCK_GROUP (MINSTREL_HT_GROUP_0 + MINSTREL_HT_GROUPS_NB)
28#define MINSTREL_VHT_GROUP_0 (MINSTREL_CCK_GROUP + 1)
29
30#define MCS_GROUP_RATES 10
31
32struct mcs_group {
33 u16 flags;
34 u8 streams;
35 u8 shift;
36 u16 duration[MCS_GROUP_RATES];
37};
38
39extern const struct mcs_group minstrel_mcs_groups[];
40
41struct minstrel_mcs_group_data {
42 u8 index;
43 u8 column;
44
45 /* sorted rate set within a MCS group*/
46 u16 max_group_tp_rate[MAX_THR_RATES];
47 u16 max_group_prob_rate;
48
49 /* MCS rate statistics */
50 struct minstrel_rate_stats rates[MCS_GROUP_RATES];
51};
52
53struct minstrel_ht_sta {
54 struct ieee80211_sta *sta;
55
56 /* ampdu length (average, per sampling interval) */
57 unsigned int ampdu_len;
58 unsigned int ampdu_packets;
59
60 /* ampdu length (EWMA) */
61 unsigned int avg_ampdu_len;
62
63 /* overall sorted rate set */
64 u16 max_tp_rate[MAX_THR_RATES];
65 u16 max_prob_rate;
66
67 /* time of last status update */
68 unsigned long last_stats_update;
69
70 /* overhead time in usec for each frame */
71 unsigned int overhead;
72 unsigned int overhead_rtscts;
73
74 unsigned int total_packets;
75 unsigned int sample_packets;
76
77 /* tx flags to add for frames for this sta */
78 u32 tx_flags;
79
80 u8 sample_wait;
81 u8 sample_tries;
82 u8 sample_count;
83 u8 sample_slow;
84
85 /* current MCS group to be sampled */
86 u8 sample_group;
87
88 u8 cck_supported;
89 u8 cck_supported_short;
90
91 /* Bitfield of supported MCS rates of all groups */
92 u16 supported[MINSTREL_GROUPS_NB];
93
94 /* MCS rate group info and statistics */
95 struct minstrel_mcs_group_data groups[MINSTREL_GROUPS_NB];
96};
97
98struct minstrel_ht_sta_priv {
99 union {
100 struct minstrel_ht_sta ht;
101 struct minstrel_sta_info legacy;
102 };
103 void *ratelist;
104 void *sample_table;
105 bool is_ht;
106};

108void minstrel_ht_add_sta_debugfs(void *priv, void *priv_sta, struct dentry *dir);
109int minstrel_ht_get_tp_avg(struct minstrel_ht_sta *mi, int group, int rate,
110 int prob_ewma);

112#endif
