27struct dst_entry *inet6_csk_route_req(const struct sock *sk,
28 struct flowi6 *fl6,
29 const struct request_sock *req,
30 u8 proto)
31{
32 struct inet_request_sock *ireq = inet_rsk(req);
33 const struct ipv6_pinfo *np = inet6_sk(sk);
34 struct in6_addr *final_p, final;
35 struct dst_entry *dst;
36
37 memset(fl6, 0, sizeof(*fl6));
38 fl6->flowi6_proto = proto;
39 fl6->daddr = ireq->ir_v6_rmt_addr;
40 rcu_read_lock();
41 final_p = fl6_update_dst(fl6, rcu_dereference(np->opt), &final);
42 rcu_read_unlock();
43 fl6->saddr = ireq->ir_v6_loc_addr;
44 fl6->flowi6_oif = ireq->ir_iif;
45 fl6->flowi6_mark = ireq->ir_mark;
46 fl6->fl6_dport = ireq->ir_rmt_port;
47 fl6->fl6_sport = htons(ireq->ir_num);
48 fl6->flowi6_uid = sk->sk_uid;
49 security_req_classify_flow(req, flowi6_to_flowi(fl6));
50
51 dst = ip6_dst_lookup_flow(sk, fl6, final_p);
52 if (IS_ERR(dst))
53 return NULL;
54
55 return dst;
56}
57EXPORT_SYMBOL(inet6_csk_route_req);

59void inet6_csk_addr2sockaddr(struct sock *sk, struct sockaddr *uaddr)
60{
61 struct sockaddr_in6 *sin6 = (struct sockaddr_in6 *) uaddr;
62
63 sin6->sin6_family = AF_INET6;
64 sin6->sin6_addr = sk->sk_v6_daddr;
65 sin6->sin6_port = inet_sk(sk)->inet_dport;
66 /* We do not store received flowlabel for TCP */
67 sin6->sin6_flowinfo = 0;
68 sin6->sin6_scope_id = ipv6_iface_scope_id(&sin6->sin6_addr,
69 sk->sk_bound_dev_if);
70}
71EXPORT_SYMBOL_GPL(inet6_csk_addr2sockaddr);

79static struct dst_entry *inet6_csk_route_socket(struct sock *sk,
80 struct flowi6 *fl6)
81{
82 struct inet_sock *inet = inet_sk(sk);
83 struct ipv6_pinfo *np = inet6_sk(sk);
84 struct in6_addr *final_p, final;
85 struct dst_entry *dst;
86
87 memset(fl6, 0, sizeof(*fl6));
88 fl6->flowi6_proto = sk->sk_protocol;
89 fl6->daddr = sk->sk_v6_daddr;
90 fl6->saddr = np->saddr;
91 fl6->flowlabel = np->flow_label;
92 IP6_ECN_flow_xmit(sk, fl6->flowlabel);
93 fl6->flowi6_oif = sk->sk_bound_dev_if;
94 fl6->flowi6_mark = sk->sk_mark;
95 fl6->fl6_sport = inet->inet_sport;
96 fl6->fl6_dport = inet->inet_dport;
97 fl6->flowi6_uid = sk->sk_uid;
98 security_sk_classify_flow(sk, flowi6_to_flowi(fl6));
99
100 rcu_read_lock();
101 final_p = fl6_update_dst(fl6, rcu_dereference(np->opt), &final);
102 rcu_read_unlock();
103
104 dst = __inet6_csk_dst_check(sk, np->dst_cookie);
105 if (!dst) {
106 dst = ip6_dst_lookup_flow(sk, fl6, final_p);
107
108 if (!IS_ERR(dst))
109 ip6_dst_store(sk, dst, NULL, NULL);
110 }
111 return dst;
112}

114int inet6_csk_xmit(struct sock *sk, struct sk_buff *skb, struct flowi *fl_unused)
115{
116 struct ipv6_pinfo *np = inet6_sk(sk);
117 struct flowi6 fl6;
118 struct dst_entry *dst;
119 int res;
120
121 dst = inet6_csk_route_socket(sk, &fl6);
122 if (IS_ERR(dst)) {
123 sk->sk_err_soft = -PTR_ERR(dst);
124 sk->sk_route_caps = 0;
125 kfree_skb(skb);
126 return PTR_ERR(dst);
127 }
128
129 rcu_read_lock();
130 skb_dst_set_noref(skb, dst);
131
132 /* Restore final destination back after routing done */
133 fl6.daddr = sk->sk_v6_daddr;
134
135 res = ip6_xmit(sk, skb, &fl6, sk->sk_mark, rcu_dereference(np->opt),
136 np->tclass);
137 rcu_read_unlock();
138 return res;
139}
140EXPORT_SYMBOL_GPL(inet6_csk_xmit);

142struct dst_entry *inet6_csk_update_pmtu(struct sock *sk, u32 mtu)
143{
144 struct flowi6 fl6;
145 struct dst_entry *dst = inet6_csk_route_socket(sk, &fl6);
146
147 if (IS_ERR(dst))
148 return NULL;
149 dst->ops->update_pmtu(dst, sk, NULL, mtu);
150
151 dst = inet6_csk_route_socket(sk, &fl6);
152 return IS_ERR(dst) ? NULL : dst;
153}
154EXPORT_SYMBOL_GPL(inet6_csk_update_pmtu);
