40static inline int sock_is_loopback(struct sock *sk)
41{
42 struct dst_entry *dst;
43 int loopback = 0;
44 rcu_read_lock();
45 dst = rcu_dereference(sk->sk_dst_cache);
46 if (dst && dst->dev &&
47 (dst->dev->features & NETIF_F_LOOPBACK))
48 loopback = 1;
49 rcu_read_unlock();
50 return loopback;
51}
