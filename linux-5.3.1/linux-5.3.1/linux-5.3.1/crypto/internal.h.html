89static inline struct crypto_alg *crypto_alg_get(struct crypto_alg *alg)
90{
91 refcount_inc(&alg->cra_refcnt);
92 return alg;
93}
94
95static inline void crypto_alg_put(struct crypto_alg *alg)
96{
97 if (refcount_dec_and_test(&alg->cra_refcnt) && alg->cra_destroy)
98 alg->cra_destroy(alg);
99}

111static inline int crypto_is_larval(struct crypto_alg *alg)
112{
113 return alg->cra_flags & CRYPTO_ALG_LARVAL;
114}
115
116static inline int crypto_is_dead(struct crypto_alg *alg)
117{
118 return alg->cra_flags & CRYPTO_ALG_DEAD;
119}
120
121static inline int crypto_is_moribund(struct crypto_alg *alg)
122{
123 return alg->cra_flags & (CRYPTO_ALG_DEAD | CRYPTO_ALG_DYING);
124}
