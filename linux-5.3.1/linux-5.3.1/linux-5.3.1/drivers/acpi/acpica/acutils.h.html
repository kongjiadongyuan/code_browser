57/*
58 * For the iASL compiler case, the output is redirected to stderr so that
59 * any of the various ACPI errors and warnings do not appear in the output
60 * files, for either the compiler or disassembler portions of the tool.
61 */
62#ifdef ACPI_ASL_COMPILER
63
64#include <stdio.h>
65
66#define ACPI_MSG_REDIRECT_BEGIN \
67 FILE *output_file = acpi_gbl_output_file; \
68 acpi_os_redirect_output (stderr);
69
70#define ACPI_MSG_REDIRECT_END \
71 acpi_os_redirect_output (output_file);
72
73#else
74/*
75 * non-iASL case - no redirection, nothing to do
76 */
77#define ACPI_MSG_REDIRECT_BEGIN
78#define ACPI_MSG_REDIRECT_END
79#endif
