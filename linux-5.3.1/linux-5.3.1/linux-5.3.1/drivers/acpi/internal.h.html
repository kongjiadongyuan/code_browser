166struct acpi_ec {
167 acpi_handle handle;
168 u32 gpe;
169 unsigned long command_addr;
170 unsigned long data_addr;
171 bool global_lock;
172 unsigned long flags;
173 unsigned long reference_count;
174 struct mutex mutex;
175 wait_queue_head_t wait;
176 struct list_head list;
177 struct transaction *curr;
178 spinlock_t lock;
179 struct work_struct work;
180 unsigned long timestamp;
181 unsigned long nr_pending_queries;
182 bool busy_polling;
183 unsigned int polling_guard;
184};
