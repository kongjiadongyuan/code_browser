115static const struct attribute_group topology_attr_group = {
116 .attrs = default_attrs,
117 .name = "topology"
118};
119
120/* Add/Remove cpu_topology interface for CPU device */
121static int topology_add_dev(unsigned int cpu)
122{
123 struct device *dev = get_cpu_device(cpu);
124
125 return sysfs_create_group(&dev->kobj, &topology_attr_group);
126}
127
128static int topology_remove_dev(unsigned int cpu)
129{
130 struct device *dev = get_cpu_device(cpu);
131
132 sysfs_remove_group(&dev->kobj, &topology_attr_group);
133 return 0;
134}
135
136static int topology_sysfs_init(void)
137{
138 return cpuhp_setup_state(CPUHP_TOPOLOGY_PREPARE,
139 "base/topology:prepare", topology_add_dev,
140 topology_remove_dev);
141}
142
143device_initcall(topology_sysfs_init);
