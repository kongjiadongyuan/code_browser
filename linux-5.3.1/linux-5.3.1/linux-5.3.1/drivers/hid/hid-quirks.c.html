920bool hid_ignore(struct hid_device *hdev)
921{
922 int i;
923
924 if (hdev->quirks & HID_QUIRK_NO_IGNORE)
925 return false;
926 if (hdev->quirks & HID_QUIRK_IGNORE)
927 return true;
928
929 switch (hdev->vendor) {
930 case USB_VENDOR_ID_CODEMERCS:
931 /* ignore all Code Mercenaries IOWarrior devices */
932 if (hdev->product >= USB_DEVICE_ID_CODEMERCS_IOW_FIRST &&
933 hdev->product <= USB_DEVICE_ID_CODEMERCS_IOW_LAST)
934 return true;
935 break;
936 case USB_VENDOR_ID_LOGITECH:
937 if (hdev->product >= USB_DEVICE_ID_LOGITECH_HARMONY_FIRST &&
938 hdev->product <= USB_DEVICE_ID_LOGITECH_HARMONY_LAST)
939 return true;
940 /*
941 * The Keene FM transmitter USB device has the same USB ID as
942 * the Logitech AudioHub Speaker, but it should ignore the hid.
943 * Check if the name is that of the Keene device.
944 * For reference: the name of the AudioHub is
945 * "HOLTEK AudioHub Speaker".
946 */
947 if (hdev->product == USB_DEVICE_ID_LOGITECH_AUDIOHUB &&
948 !strcmp(hdev->name, "HOLTEK B-LINK USB Audio "))
949 return true;
950 break;
951 case USB_VENDOR_ID_SOUNDGRAPH:
952 if (hdev->product >= USB_DEVICE_ID_SOUNDGRAPH_IMON_FIRST &&
953 hdev->product <= USB_DEVICE_ID_SOUNDGRAPH_IMON_LAST)
954 return true;
955 break;
956 case USB_VENDOR_ID_HANWANG:
957 if (hdev->product >= USB_DEVICE_ID_HANWANG_TABLET_FIRST &&
958 hdev->product <= USB_DEVICE_ID_HANWANG_TABLET_LAST)
959 return true;
960 break;
961 case USB_VENDOR_ID_JESS:
962 if (hdev->product == USB_DEVICE_ID_JESS_YUREX &&
963 hdev->type == HID_TYPE_USBNONE)
964 return true;
965 break;
966 case USB_VENDOR_ID_VELLEMAN:
967 /* These are not HID devices. They are handled by comedi. */
968 if ((hdev->product >= USB_DEVICE_ID_VELLEMAN_K8055_FIRST &&
969 hdev->product <= USB_DEVICE_ID_VELLEMAN_K8055_LAST) ||
970 (hdev->product >= USB_DEVICE_ID_VELLEMAN_K8061_FIRST &&
971 hdev->product <= USB_DEVICE_ID_VELLEMAN_K8061_LAST))
972 return true;
973 break;
974 case USB_VENDOR_ID_ATMEL_V_USB:
975 /* Masterkit MA901 usb radio based on Atmel tiny85 chip and
976 * it has the same USB ID as many Atmel V-USB devices. This
977 * usb radio is handled by radio-ma901.c driver so we want
978 * ignore the hid. Check the name, bus, product and ignore
979 * if we have MA901 usb radio.
980 */
981 if (hdev->product == USB_DEVICE_ID_ATMEL_V_USB &&
982 hdev->bus == BUS_USB &&
983 strncmp(hdev->name, "www.masterkit.ru MA901", 22) == 0)
984 return true;
985 break;
986 case USB_VENDOR_ID_ELAN:
987 /*
988 * Blacklist of everything that gets handled by the elan_i2c
989 * input driver. This avoids disabling valid touchpads and
990 * other ELAN devices.
991 */
992 if ((hdev->product == 0x0401 || hdev->product == 0x0400))
993 for (i = 0; strlen(elan_acpi_id[i].id); ++i)
994 if (!strncmp(hdev->name, elan_acpi_id[i].id,
995 strlen(elan_acpi_id[i].id)))
996 return true;
997 break;
998 }
999
1000 if (hdev->type == HID_TYPE_USBMOUSE &&
1001 hid_match_id(hdev, hid_mouse_ignore_list))
1002 return true;
1003
1004 return !!hid_match_id(hdev, hid_ignore_list);
1005}

1009struct quirks_list_struct {

1011 struct list_head node;

1019/**
1020 * hid_exists_dquirk: find any dynamic quirks for a HID device
1021 * @hdev: the HID device to match
1022 *
1023 * Description:
1024 * Scans dquirks_list for a matching dynamic quirk and returns
1025 * the pointer to the relevant struct hid_device_id if found.
1026 * Must be called with a read lock held on dquirks_lock.
1027 *
1028 * Returns: NULL if no quirk found, struct hid_device_id * if found.
1029 */
1030static struct hid_device_id *hid_exists_dquirk(const struct hid_device *hdev)
1031{
1032 struct quirks_list_struct *q;
1033 struct hid_device_id *bl_entry = NULL;
1034
1035 list_for_each_entry(q, &dquirks_list, node) {
1036 if (hid_match_one_id(hdev, &q->hid_bl_item)) {
1037 bl_entry = &q->hid_bl_item;
1038 break;
1039 }
1040 }
1041
1042 if (bl_entry != NULL)
1043 dbg_hid("Found dynamic quirk 0x%lx for HID device 0x%04x:0x%04x\n",
1044 bl_entry->driver_data, bl_entry->vendor,
1045 bl_entry->product);
1046
1047 return bl_entry;
1048}

1051/**
1052 * hid_modify_dquirk: add/replace a HID quirk
1053 * @id: the HID device to match
1054 * @quirks: the unsigned long quirks value to add/replace
1055 *
1056 * Description:
1057 * If an dynamic quirk exists in memory for this device, replace its
1058 * quirks value with what was provided. Otherwise, add the quirk
1059 * to the dynamic quirks list.
1060 *
1061 * Returns: 0 OK, -error on failure.
1062 */
1063static int hid_modify_dquirk(const struct hid_device_id *id,
1064 const unsigned long quirks)
1065{
1066 struct hid_device *hdev;
1067 struct quirks_list_struct *q_new, *q;
1068 int list_edited = 0;
1069 int ret = 0;
1070
1071 hdev = kzalloc(sizeof(*hdev), GFP_KERNEL);
1072 if (!hdev)
1073 return -ENOMEM;
1074
1075 q_new = kmalloc(sizeof(struct quirks_list_struct), GFP_KERNEL);
1076 if (!q_new) {
1077 ret = -ENOMEM;
1078 goto out;
1079 }
1080
1081 hdev->bus = q_new->hid_bl_item.bus = id->bus;
1082 hdev->group = q_new->hid_bl_item.group = id->group;
1083 hdev->vendor = q_new->hid_bl_item.vendor = id->vendor;
1084 hdev->product = q_new->hid_bl_item.product = id->product;
1085 q_new->hid_bl_item.driver_data = quirks;
1086
1087 mutex_lock(&dquirks_lock);
1088
1089 list_for_each_entry(q, &dquirks_list, node) {
1090
1091 if (hid_match_one_id(hdev, &q->hid_bl_item)) {
1092
1093 list_replace(&q->node, &q_new->node);
1094 kfree(q);
1095 list_edited = 1;
1096 break;
1097
1098 }
1099
1100 }
1101
1102 if (!list_edited)
1103 list_add_tail(&q_new->node, &dquirks_list);
1104
1105 mutex_unlock(&dquirks_lock);
1106
1107 out:
1108 kfree(hdev);
1109 return ret;
1110}

1137 * hid_quirks_init: apply HID quirks specified at module load time

1139int hid_quirks_init(char **quirks_param, __u16 bus, int count)
1140{
1141 struct hid_device_id id = { 0 };

1143 unsigned short int vendor, product;
1144 u32 quirks;
1145
1146 id.bus = bus;

1185 * hid_gets_squirk: return any static quirks for a HID device
1186 * @hdev: the HID device to match
1187 *
1188 * Description:
1189 * Given a HID device, return a pointer to the quirked hid_device_id entry
1190 * associated with that device.
1191 *
1192 * Returns: the quirks.
1193 */
1194static unsigned long hid_gets_squirk(const struct hid_device *hdev)
1195{

1197 unsigned long quirks = 0;
1198
1199 if (hid_match_id(hdev, hid_ignore_list))
1200 quirks |= HID_QUIRK_IGNORE;

1203 quirks |= HID_QUIRK_HAVE_SPECIAL_DRIVER;

1206 if (bl_entry != NULL)

1208
1209 if (quirks)

1211 quirks, hdev->vendor, hdev->product);

1215/**
1216 * hid_lookup_quirk: return any quirks associated with a HID device
1217 * @hdev: the HID device to look for
1218 *
1219 * Description:
1220 * Given a HID device, return any quirks associated with that device.
1221 *
1222 * Returns: an unsigned long quirks value.
1223 */
1224unsigned long hid_lookup_quirk(const struct hid_device *hdev)
1225{
1226 unsigned long quirks = 0;
1227 const struct hid_device_id *quirk_entry = NULL;
1228
1229 /* NCR devices must not be queried for reports */
1230 if (hdev->bus == BUS_USB &&
1231 hdev->vendor == USB_VENDOR_ID_NCR &&
1232 hdev->product >= USB_DEVICE_ID_NCR_FIRST &&
1233 hdev->product <= USB_DEVICE_ID_NCR_LAST)
1234 return HID_QUIRK_NO_INIT_REPORTS;
1235
1236 /* These devices must be ignored if version (bcdDevice) is too old */
1237 if (hdev->bus == BUS_USB && hdev->vendor == USB_VENDOR_ID_JABRA) {
1238 switch (hdev->product) {
1239 case USB_DEVICE_ID_JABRA_SPEAK_410:
1240 if (hdev->version < 0x0111)
1241 return HID_QUIRK_IGNORE;
1242 break;
1243 case USB_DEVICE_ID_JABRA_SPEAK_510:
1244 if (hdev->version < 0x0214)
1245 return HID_QUIRK_IGNORE;
1246 break;
1247 }
1248 }
1249
1250 mutex_lock(&dquirks_lock);
1251 quirk_entry = hid_exists_dquirk(hdev);
1252 if (quirk_entry)
1253 quirks = quirk_entry->driver_data;
1254 else
1255 quirks = hid_gets_squirk(hdev);
1256 mutex_unlock(&dquirks_lock);
1257
1258 return quirks;
1259}
