29struct drm_i915_private;
30struct drm_i915_gem_object;

32enum fb_op_origin {
33 ORIGIN_GTT,
34 ORIGIN_CPU,
35 ORIGIN_CS,
36 ORIGIN_FLIP,
37 ORIGIN_DIRTYFB,
38};

40void intel_frontbuffer_flip_prepare(struct drm_i915_private *dev_priv,

47void __intel_fb_obj_invalidate(struct drm_i915_gem_object *obj,
48 enum fb_op_origin origin,
49 unsigned int frontbuffer_bits);
50void __intel_fb_obj_flush(struct drm_i915_gem_object *obj,
51 enum fb_op_origin origin,
52 unsigned int frontbuffer_bits);

54/**
55 * intel_fb_obj_invalidate - invalidate frontbuffer object
56 * @obj: GEM object to invalidate
57 * @origin: which operation caused the invalidation
58 *
59 * This function gets called every time rendering on the given object starts and
60 * frontbuffer caching (fbc, low refresh rate for DRRS, panel self refresh) must
61 * be invalidated. For ORIGIN_CS any subsequent invalidation will be delayed
62 * until the rendering completes or a flip on this frontbuffer plane is
63 * scheduled.
64 */
65static inline bool intel_fb_obj_invalidate(struct drm_i915_gem_object *obj,
66 enum fb_op_origin origin)
67{
68 unsigned int frontbuffer_bits;
69
70 frontbuffer_bits = atomic_read(&obj->frontbuffer_bits);
71 if (!frontbuffer_bits)
72 return false;
73
74 __intel_fb_obj_invalidate(obj, origin, frontbuffer_bits);
75 return true;
76}
77
78/**
79 * intel_fb_obj_flush - flush frontbuffer object
80 * @obj: GEM object to flush
81 * @origin: which operation caused the flush
82 *
83 * This function gets called every time rendering on the given object has
84 * completed and frontbuffer caching can be started again.
85 */
86static inline void intel_fb_obj_flush(struct drm_i915_gem_object *obj,
87 enum fb_op_origin origin)
88{
89 unsigned int frontbuffer_bits;
90
91 frontbuffer_bits = atomic_read(&obj->frontbuffer_bits);
92 if (!frontbuffer_bits)
93 return;
94
95 __intel_fb_obj_flush(obj, origin, frontbuffer_bits);
96}
