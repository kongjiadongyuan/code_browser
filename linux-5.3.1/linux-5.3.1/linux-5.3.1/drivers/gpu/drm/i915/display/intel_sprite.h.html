37static inline bool icl_is_nv12_y_plane(enum plane_id id)
38{
39 /* Don't need to do a gen check, these planes are only available on gen11 */
40 if (id == PLANE_SPRITE4 || id == PLANE_SPRITE5)
41 return true;
42
43 return false;
44}
45
46static inline u8 icl_hdr_plane_mask(void)
47{
48 return BIT(PLANE_PRIMARY) |
49 BIT(PLANE_SPRITE0) | BIT(PLANE_SPRITE1);
50}
51
52static inline bool icl_is_hdr_plane(struct drm_i915_private *dev_priv,
53 enum plane_id plane_id)
54{
55 return INTEL_GEN(dev_priv) >= 11 &&
56 icl_hdr_plane_mask() & BIT(plane_id);
57}
