25#ifndef _I915_PARAMS_H_
26#define _I915_PARAMS_H_
27
28#include <linux/bitops.h>
29#include <linux/cache.h> /* for __read_mostly */
30
31struct drm_printer;
32
33#define ENABLE_GUC_SUBMISSION BIT(0)
34#define ENABLE_GUC_LOAD_HUC BIT(1)
35
36/*
37 * Invoke param, a function-like macro, for each i915 param, with arguments:
38 *
39 * param(type, name, value)
40 *
41 * type: parameter type, one of {bool, int, unsigned int, char *}
42 * name: name of the parameter
43 * value: initial/default value of the parameter
44 */
45#define I915_PARAMS_FOR_EACH(param) \
46 param(char *, vbt_firmware, NULL) \
47 param(int, modeset, -1) \
48 param(int, lvds_channel_mode, 0) \
49 param(int, panel_use_ssc, -1) \
50 param(int, vbt_sdvo_panel_type, -1) \
51 param(int, enable_dc, -1) \
52 param(int, enable_fbc, -1) \
53 param(int, enable_psr, -1) \
54 param(int, disable_power_well, -1) \
55 param(int, enable_ips, 1) \
56 param(int, invert_brightness, 0) \
57 param(int, enable_guc, 0) \
58 param(int, guc_log_level, -1) \
59 param(char *, guc_firmware_path, NULL) \
60 param(char *, huc_firmware_path, NULL) \
61 param(char *, dmc_firmware_path, NULL) \
62 param(int, mmio_debug, -IS_ENABLED(CONFIG_DRM_I915_DEBUG_MMIO)) \
63 param(int, edp_vswing, 0) \
64 param(int, reset, 2) \
65 param(unsigned int, inject_load_failure, 0) \
66 param(int, fastboot, -1) \
67 param(char *, force_probe, CONFIG_DRM_I915_FORCE_PROBE) \
68 /* leave bools at the end to not create holes */ \
69 param(bool, alpha_support, IS_ENABLED(CONFIG_DRM_I915_ALPHA_SUPPORT)) \
70 param(bool, enable_hangcheck, true) \
71 param(bool, prefault_disable, false) \
72 param(bool, load_detect_test, false) \
73 param(bool, force_reset_modeset_test, false) \
74 param(bool, error_capture, true) \
75 param(bool, disable_display, false) \
76 param(bool, verbose_state_checks, true) \
77 param(bool, nuclear_pageflip, false) \
78 param(bool, enable_dp_mst, true) \
79 param(bool, enable_dpcd_backlight, false) \
80 param(bool, enable_gvt, false)
81
82#define MEMBER(T, member, ...) T member;
83struct i915_params {
84 I915_PARAMS_FOR_EACH(MEMBER);
85};
86#undef MEMBER
87
88extern struct i915_params i915_modparams __read_mostly;
89
90void i915_params_dump(const struct i915_params *params, struct drm_printer *p);
91void i915_params_copy(struct i915_params *dest, const struct i915_params *src);
92void i915_params_free(struct i915_params *params);
93
94#endif
