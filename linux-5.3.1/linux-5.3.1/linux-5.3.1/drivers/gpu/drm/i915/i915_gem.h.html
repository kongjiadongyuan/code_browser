37#define GEM_BUG_ON(condition) do { if (unlikely((condition))) { \
38 pr_err("%s:%d GEM_BUG_ON(%s)\n", \

78static inline void __tasklet_disable_sync_once(struct tasklet_struct *t)
79{
80 if (!atomic_fetch_inc(&t->count))
81 tasklet_unlock_wait(t);
82}
83
84static inline bool __tasklet_is_enabled(const struct tasklet_struct *t)
85{
86 return !atomic_read(&t->count);
87}
88
89static inline bool __tasklet_enable(struct tasklet_struct *t)
90{
91 return atomic_dec_and_test(&t->count);
92}
93
94static inline bool __tasklet_is_scheduled(struct tasklet_struct *t)
95{
96 return test_bit(TASKLET_STATE_SCHED, &t->state);
97}
