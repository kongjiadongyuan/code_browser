46static inline bool intel_uc_is_using_guc(struct drm_i915_private *i915)
47{
48 GEM_BUG_ON(i915_modparams.enable_guc < 0);
49 return i915_modparams.enable_guc > 0;
50}
51
52static inline bool intel_uc_is_using_guc_submission(struct drm_i915_private *i915)
53{
54 GEM_BUG_ON(i915_modparams.enable_guc < 0);
55 return i915_modparams.enable_guc & ENABLE_GUC_SUBMISSION;
56}
57
58static inline bool intel_uc_is_using_huc(struct drm_i915_private *i915)
59{
60 GEM_BUG_ON(i915_modparams.enable_guc < 0);
61 return i915_modparams.enable_guc & ENABLE_GUC_LOAD_HUC;
62}
