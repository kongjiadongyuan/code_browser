47/* We extract the function declarations from i915_mock_selftests.h and
48 * i915_live_selftests.h Add your unit test declarations there!
49 *
50 * Mock unit tests are run very early upon module load, before the driver
51 * is probed. All hardware interactions, as well as other subsystems, must
52 * be "mocked".
53 *
54 * Live unit tests are run after the driver is loaded - all hardware
55 * interactions are real.
56 */
57#define selftest(name, func) int func(void);
58#include "selftests/i915_mock_selftests.h"
59#undef selftest
60#define selftest(name, func) int func(struct drm_i915_private *i915);
61#include "selftests/i915_live_selftests.h"
62#undef selftest
63
64struct i915_subtest {
65 int (*func)(void *data);
66 const char *name;
67};
68
69int __i915_subtests(const char *caller,
70 const struct i915_subtest *st,
71 unsigned int count,
72 void *data);
73#define i915_subtests(T, data) \
74 __i915_subtests(__func__, T, ARRAY_SIZE(T), data)
75
76#define SUBTEST(x) { x, #x }

78#define I915_SELFTEST_DECLARE(x) x
79#define I915_SELFTEST_ONLY(x) unlikely(x)

91/* Using the i915_selftest_ prefix becomes a little unwieldy with the helpers.
92 * Instead we use the igt_ shorthand, in reference to the intel-gpu-tools
93 * suite of uabi test cases (which includes a test runner for our selftests).
94 */
95
96#define IGT_TIMEOUT(name__) \
97 unsigned long name__ = jiffies + i915_selftest.timeout_jiffies

99__printf(2, 3)
100bool __igt_timeout(unsigned long timeout, const char *fmt, ...);

102#define igt_timeout(t, fmt, ...) \
103 __igt_timeout((t), KERN_DEBUG pr_fmt(fmt), ##__VA_ARGS__)
