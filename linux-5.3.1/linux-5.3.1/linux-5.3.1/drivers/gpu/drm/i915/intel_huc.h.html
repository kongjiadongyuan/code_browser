25#ifndef _INTEL_HUC_H_
26#define _INTEL_HUC_H_
27
28#include "i915_reg.h"
29#include "intel_uc_fw.h"
30#include "intel_huc_fw.h"
31
32struct intel_huc {
33 /* Generic uC firmware management */
34 struct intel_uc_fw fw;
35
36 /* HuC-specific additions */
37 struct i915_vma *rsa_data;
38 void *rsa_data_vaddr;
39
40 struct {
41 i915_reg_t reg;
42 u32 mask;
43 u32 value;
44 } status;
45};
46
47void intel_huc_init_early(struct intel_huc *huc);
48int intel_huc_init_misc(struct intel_huc *huc);
49int intel_huc_init(struct intel_huc *huc);
50void intel_huc_fini(struct intel_huc *huc);
51int intel_huc_auth(struct intel_huc *huc);
52int intel_huc_check_status(struct intel_huc *huc);
53
54static inline void intel_huc_fini_misc(struct intel_huc *huc)
55{
56 intel_uc_fw_cleanup_fetch(&huc->fw);
57}
58
59static inline int intel_huc_sanitize(struct intel_huc *huc)
60{
61 intel_uc_fw_sanitize(&huc->fw);
62 return 0;
63}
64
65#endif
