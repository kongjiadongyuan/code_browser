29#include <linux/compat.h>

31#include <drm/i915_drm.h>
32#include <drm/drm_ioctl.h>
33#include "i915_drv.h"
34
35struct drm_i915_getparam32 {
36 s32 param;
37 /*
38 * We screwed up the generic ioctl struct here and used a variable-sized
39 * pointer. Use u32 in the compat struct to match the 32bit pointer
40 * userspace expects.
41 */
42 u32 value;
43};
44
45static int compat_i915_getparam(struct file *file, unsigned int cmd,
46 unsigned long arg)
47{
48 struct drm_i915_getparam32 req32;
49 drm_i915_getparam_t __user *request;
50
51 if (copy_from_user(&req32, (void __user *)arg, sizeof(req32)))
52 return -EFAULT;
53
54 request = compat_alloc_user_space(sizeof(*request));
55 if (!access_ok(request, sizeof(*request)) ||
56 __put_user(req32.param, &request->param) ||
57 __put_user((void __user *)(unsigned long)req32.value,
58 &request->value))
59 return -EFAULT;
60
61 return drm_ioctl(file, DRM_IOCTL_I915_GETPARAM,
62 (unsigned long)request);
63}
64
65static drm_ioctl_compat_t *i915_compat_ioctls[] = {
66 [DRM_I915_GETPARAM] = compat_i915_getparam,
67};
68
69/**
70 * i915_compat_ioctl - handle the mistakes of the past
71 * @filp: the file pointer
72 * @cmd: the ioctl command (and encoded flags)
73 * @arg: the ioctl argument (from userspace)
74 *
75 * Called whenever a 32-bit process running under a 64-bit kernel
76 * performs an ioctl on /dev/dri/card<n>.
77 */
78long i915_compat_ioctl(struct file *filp, unsigned int cmd, unsigned long arg)
79{
80 unsigned int nr = DRM_IOCTL_NR(cmd);
81 drm_ioctl_compat_t *fn = NULL;
82 int ret;
83
84 if (nr < DRM_COMMAND_BASE || nr >= DRM_COMMAND_END)
85 return drm_compat_ioctl(filp, cmd, arg);
86
87 if (nr < DRM_COMMAND_BASE + ARRAY_SIZE(i915_compat_ioctls))
88 fn = i915_compat_ioctls[nr - DRM_COMMAND_BASE];
89
90 if (fn != NULL)
91 ret = (*fn) (filp, cmd, arg);
92 else
93 ret = drm_ioctl(filp, cmd, arg);
94
95 return ret;
96}
