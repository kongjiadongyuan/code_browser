843static void i915_pci_remove(struct pci_dev *pdev)
844{
845 struct drm_device *dev;
846
847 dev = pci_get_drvdata(pdev);
848 if (!dev) /* driver load aborted, nothing to cleanup */
849 return;
850
851 i915_driver_unload(dev);
852 drm_dev_put(dev);
853
854 pci_set_drvdata(pdev, NULL);
855}

857/* is device_id present in comma separated list of ids */
858static bool force_probe(u16 device_id, const char *devices)
859{
860 char *s, *p, *tok;
861 bool ret;
862
863 /* FIXME: transitional */
864 if (i915_modparams.alpha_support) {
865 DRM_INFO("i915.alpha_support is deprecated, use i915.force_probe=%04x instead\n",
866 device_id);
867 return true;
868 }
869
870 if (!devices || !*devices)
871 return false;
872
873 /* match everything */
874 if (strcmp(devices, "*") == 0)
875 return true;
876
877 s = kstrdup(devices, GFP_KERNEL);
878 if (!s)
879 return false;
880
881 for (p = s, ret = false; (tok = strsep(&p, ",")) != NULL; ) {
882 u16 val;
883
884 if (kstrtou16(tok, 16, &val) == 0 && val == device_id) {
885 ret = true;
886 break;
887 }
888 }
889
890 kfree(s);
891
892 return ret;
893}

895static int i915_pci_probe(struct pci_dev *pdev, const struct pci_device_id *ent)
896{
897 struct intel_device_info *intel_info =
898 (struct intel_device_info *) ent->driver_data;
899 int err;
900
901 if (intel_info->require_force_probe &&
902 !force_probe(pdev->device, i915_modparams.force_probe)) {
903 DRM_INFO("Your graphics device %04x is not properly supported by the driver in this\n"
904 "kernel version. To force driver probe anyway, use i915.force_probe=%04x\n"
905 "module parameter or CONFIG_DRM_I915_FORCE_PROBE=%04x configuration option,\n"
906 "or (recommended) check for kernel updates.\n",
907 pdev->device, pdev->device, pdev->device);
908 return -ENODEV;
909 }
910
911 /* Only bind to function 0 of the device. Early generations
912 * used function 1 as a placeholder for multi-head. This causes
913 * us confusion instead, especially on the systems where both
914 * functions have the same PCI-ID!
915 */
916 if (PCI_FUNC(pdev->devfn))
917 return -ENODEV;
918
919 /*
920 * apple-gmux is needed on dual GPU MacBook Pro
921 * to probe the panel if we're the inactive GPU.
922 */
923 if (vga_switcheroo_client_probe_defer(pdev))
924 return -EPROBE_DEFER;
925
926 err = i915_driver_load(pdev, ent);
927 if (err)
928 return err;
929
930 if (i915_inject_load_failure()) {
931 i915_pci_remove(pdev);
932 return -ENODEV;
933 }
934
935 err = i915_live_selftests(pdev);
936 if (err) {
937 i915_pci_remove(pdev);
938 return err > 0 ? -ENOTTY : err;
939 }
940
941 return 0;
942}

952static int __init i915_init(void)
953{
954 bool use_kms = true;
955 int err;
956
957 err = i915_globals_init();
958 if (err)
959 return err;
960
961 err = i915_mock_selftests();
962 if (err)
963 return err > 0 ? 0 : err;
964
965 /*
966 * Enable KMS by default, unless explicitly overriden by
967 * either the i915.modeset prarameter or by the
968 * vga_text_mode_force boot option.
969 */
970
971 if (i915_modparams.modeset == 0)
972 use_kms = false;
973
974 if (vgacon_text_force() && i915_modparams.modeset == -1)
975 use_kms = false;
976
977 if (!use_kms) {
978 /* Silently fail loading to not upset userspace. */
979 DRM_DEBUG_DRIVER("KMS disabled.\n");
980 return 0;
981 }
982
983 return pci_register_driver(&i915_pci_driver);
984}

989 return;
990
991 pci_unregister_driver(&i915_pci_driver);
992 i915_globals_exit();
993}
