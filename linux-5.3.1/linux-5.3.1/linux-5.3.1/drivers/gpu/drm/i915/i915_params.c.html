180static __always_inline void _print_param(struct drm_printer *p,
181 const char *name,
182 const char *type,
183 const void *x)
184{
185 if (!__builtin_strcmp(type, "bool"))
186 drm_printf(p, "i915.%s=%s\n", name, yesno(*(const bool *)x));
187 else if (!__builtin_strcmp(type, "int"))
188 drm_printf(p, "i915.%s=%d\n", name, *(const int *)x);
189 else if (!__builtin_strcmp(type, "unsigned int"))
190 drm_printf(p, "i915.%s=%u\n", name, *(const unsigned int *)x);
191 else if (!__builtin_strcmp(type, "char *"))
192 drm_printf(p, "i915.%s=%s\n", name, *(const char **)x);
193 else
194 WARN_ONCE(1, "no printer defined for param type %s (i915.%s)\n",
195 type, name);
196}

198/**
199 * i915_params_dump - dump i915 modparams
200 * @params: i915 modparams
201 * @p: the &drm_printer
202 *
203 * Pretty printer for i915 modparams.
204 */
205void i915_params_dump(const struct i915_params *params, struct drm_printer *p)
206{
207#define PRINT(T, x, ...) _print_param(p, #x, #T, &params->x);
208 I915_PARAMS_FOR_EACH(PRINT);
209#undef PRINT
210}

212static __always_inline void dup_param(const char *type, void *x)
213{
214 if (!__builtin_strcmp(type, "char *"))
215 *(void **)x = kstrdup(*(void **)x, GFP_ATOMIC);
216}
217
218void i915_params_copy(struct i915_params *dest, const struct i915_params *src)
219{
220 *dest = *src;
221#define DUP(T, x, ...) dup_param(#T, &dest->x);
222 I915_PARAMS_FOR_EACH(DUP);
223#undef DUP
224}
225
226static __always_inline void free_param(const char *type, void *x)
227{
228 if (!__builtin_strcmp(type, "char *")) {
229 kfree(*(void **)x);
230 *(void **)x = NULL;
231 }
232}

234/* free the allocated members, *not* the passed in params itself */
235void i915_params_free(struct i915_params *params)
236{
237#define FREE(T, x, ...) free_param(#T, &params->x);
238 I915_PARAMS_FOR_EACH(FREE);
239#undef FREE
240}
