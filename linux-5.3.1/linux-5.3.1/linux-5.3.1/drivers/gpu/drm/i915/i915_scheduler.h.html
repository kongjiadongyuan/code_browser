55static inline bool i915_scheduler_need_preempt(int prio, int active)
56{
57 /*
58 * Allow preemption of low -> normal -> high, but we do
59 * not allow low priority tasks to preempt other low priority
60 * tasks under the impression that latency for low priority
61 * tasks does not matter (as much as background throughput),
62 * so kiss.
63 *
64 * More naturally we would write
65 * prio >= max(0, last);
66 * except that we wish to prevent triggering preemption at the same
67 * priority level: the task that is running should remain running
68 * to preserve FIFO ordering of dependencies.
69 */
70 return prio > max(I915_PRIORITY_NORMAL - 1, active);
71}
