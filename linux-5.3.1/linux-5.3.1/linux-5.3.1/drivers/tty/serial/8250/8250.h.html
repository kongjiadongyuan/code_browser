73#define UART_CAP_FIFO (1 << 8) /* UART has FIFO */
74#define UART_CAP_EFR (1 << 9) /* UART has EFR */
75#define UART_CAP_SLEEP (1 << 10) /* UART has IER sleep */
76#define UART_CAP_AFE (1 << 11) /* MCR-based hw flow control */
77#define UART_CAP_UUE (1 << 12) /* UART needs IER bit 6 set (Xscale) */
78#define UART_CAP_RTOIE (1 << 13) /* UART needs IER bit 4 set (Xscale, Tegra) */
79#define UART_CAP_HFIFO (1 << 14) /* UART has a "hidden" FIFO */
80#define UART_CAP_RPM (1 << 15) /* Runtime PM is active while idle */
81#define UART_CAP_IRDA (1 << 16) /* UART supports IrDA line discipline */
82#define UART_CAP_MINI (1 << 17) /* Mini UART on BCM283X family lacks:
83 * STOP PARITY EPAR SPAR WLEN5 WLEN6
84 */
85
86#define UART_BUG_QUOT (1 << 0) /* UART has buggy quot LSB */
87#define UART_BUG_TXEN (1 << 1) /* UART has buggy TX IIR status */
88#define UART_BUG_NOMSR (1 << 2) /* UART has buggy MSR status bits (Au1x00) */
89#define UART_BUG_THRE (1 << 3) /* UART has buggy THRE reassertion */
90#define UART_BUG_PARITY (1 << 4) /* UART mishandles parity if FIFO enabled */

133static inline bool serial8250_set_THRI(struct uart_8250_port *up)
134{
135 if (up->ier & UART_IER_THRI)
136 return false;
137 up->ier |= UART_IER_THRI;
138 serial_out(up, UART_IER, up->ier);
139 return true;
140}

142static inline bool serial8250_clear_THRI(struct uart_8250_port *up)
143{
144 if (!(up->ier & UART_IER_THRI))
145 return false;
146 up->ier &= ~UART_IER_THRI;
147 serial_out(up, UART_IER, up->ier);
148 return true;
149}

162/* MCR <-> TIOCM conversion */
163static inline int serial8250_TIOCM_to_MCR(int tiocm)
164{
165 int mcr = 0;
166
167 if (tiocm & TIOCM_RTS)
168 mcr |= UART_MCR_RTS;
169 if (tiocm & TIOCM_DTR)
170 mcr |= UART_MCR_DTR;
171 if (tiocm & TIOCM_OUT1)
172 mcr |= UART_MCR_OUT1;
173 if (tiocm & TIOCM_OUT2)
174 mcr |= UART_MCR_OUT2;
175 if (tiocm & TIOCM_LOOP)
176 mcr |= UART_MCR_LOOP;
177
178 return mcr;
179}
180
181static inline int serial8250_MCR_to_TIOCM(int mcr)
182{
183 int tiocm = 0;
184
185 if (mcr & UART_MCR_RTS)
186 tiocm |= TIOCM_RTS;
187 if (mcr & UART_MCR_DTR)
188 tiocm |= TIOCM_DTR;
189 if (mcr & UART_MCR_OUT1)
190 tiocm |= TIOCM_OUT1;
191 if (mcr & UART_MCR_OUT2)
192 tiocm |= TIOCM_OUT2;
193 if (mcr & UART_MCR_LOOP)
194 tiocm |= TIOCM_LOOP;
195
196 return tiocm;
197}
198
199/* MSR <-> TIOCM conversion */
200static inline int serial8250_MSR_to_TIOCM(int msr)
201{
202 int tiocm = 0;
203
204 if (msr & UART_MSR_DCD)
205 tiocm |= TIOCM_CAR;
206 if (msr & UART_MSR_RI)
207 tiocm |= TIOCM_RNG;
208 if (msr & UART_MSR_DSR)
209 tiocm |= TIOCM_DSR;
210 if (msr & UART_MSR_CTS)
211 tiocm |= TIOCM_CTS;
212
213 return tiocm;
214}

216static inline void serial8250_out_MCR(struct uart_8250_port *up, int value)
217{
218 serial_out(up, UART_MCR, value);
219
220 if (up->gpios)
221 mctrl_gpio_set(up->gpios, serial8250_MCR_to_TIOCM(value));
222}
223
224static inline int serial8250_in_MCR(struct uart_8250_port *up)
225{
226 int mctrl;
227
228 mctrl = serial_in(up, UART_MCR);
229
230 if (up->gpios) {
231 unsigned int mctrl_gpio = 0;
232
233 mctrl_gpio = mctrl_gpio_get_outputs(up->gpios, &mctrl_gpio);
234 mctrl |= serial8250_TIOCM_to_MCR(mctrl_gpio);
235 }
236
237 return mctrl;
238}

325static inline int ns16550a_goto_highspeed(struct uart_8250_port *up)
326{
327 unsigned char status;
328
329 status = serial_in(up, 0x04); /* EXCR2 */
330#define PRESL(x) ((x) & 0x30)
331 if (PRESL(status) == 0x10) {
332 /* already in high speed mode */
333 return 0;
334 } else {
335 status &= ~0xB0; /* Disable LOCK, mask out PRESL[01] */
336 status |= 0x10; /* 1.625 divisor for baud_base --> 921600 */
337 serial_out(up, 0x04, status);
338 }
339 return 1;
340}
