14/**
15 * blk_mq_pci_map_queues - provide a default queue mapping for PCI device
16 * @qmap: CPU to hardware queue map.
17 * @pdev: PCI device associated with @set.
18 * @offset: Offset to use for the pci irq vector
19 *
20 * This function assumes the PCI device @pdev has at least as many available
21 * interrupt vectors as @set has queues. It will then query the vector
22 * corresponding to each queue for it's affinity mask and built queue mapping
23 * that maps a queue to the CPUs that have irq affinity for the corresponding
24 * vector.
25 */
26int blk_mq_pci_map_queues(struct blk_mq_queue_map *qmap, struct pci_dev *pdev,
27 int offset)
28{
29 const struct cpumask *mask;
30 unsigned int queue, cpu;
31
32 for (queue = 0; queue < qmap->nr_queues; queue++) {
33 mask = pci_irq_get_affinity(pdev, queue + offset);
34 if (!mask)
35 goto fallback;
36
37 for_each_cpu(cpu, mask)
38 qmap->mq_map[cpu] = qmap->queue_offset + queue;
39 }
40
41 return 0;
42
43fallback:
44 WARN_ON_ONCE(qmap->nr_queues > 1);
45 blk_mq_clear_mq_map(qmap);
46 return 0;
47}
48EXPORT_SYMBOL_GPL(blk_mq_pci_map_queues);
