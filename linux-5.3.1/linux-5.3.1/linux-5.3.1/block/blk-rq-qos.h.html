57static inline struct rq_qos *rq_qos_id(struct request_queue *q,
58 enum rq_qos_id id)
59{
60 struct rq_qos *rqos;
61 for (rqos = q->rq_qos; rqos; rqos = rqos->next) {
62 if (rqos->id == id)
63 break;
64 }
65 return rqos;
66}
67
68static inline struct rq_qos *wbt_rq_qos(struct request_queue *q)
69{
70 return rq_qos_id(q, RQ_QOS_WBT);
71}
72
73static inline struct rq_qos *blkcg_rq_qos(struct request_queue *q)
74{
75 return rq_qos_id(q, RQ_QOS_CGROUP);
76}

78static inline const char *rq_qos_id_to_name(enum rq_qos_id id)
79{
80 switch (id) {
81 case RQ_QOS_WBT:
82 return "wbt";
83 case RQ_QOS_CGROUP:
84 return "cgroup";
85 }
86 return "unknown";
87}

89static inline void rq_wait_init(struct rq_wait *rq_wait)
90{
91 atomic_set(&rq_wait->inflight, 0);
92 init_waitqueue_head(&rq_wait->wait);
93}

95static inline void rq_qos_add(struct request_queue *q, struct rq_qos *rqos)
96{
97 rqos->next = q->rq_qos;
98 q->rq_qos = rqos;
99
100 if (rqos->ops->debugfs_attrs)
101 blk_mq_debugfs_register_rqos(rqos);
102}
103
104static inline void rq_qos_del(struct request_queue *q, struct rq_qos *rqos)
105{
106 struct rq_qos *cur, *prev = NULL;
107 for (cur = q->rq_qos; cur; cur = cur->next) {
108 if (cur == rqos) {
109 if (prev)
110 prev->next = rqos->next;
111 else
112 q->rq_qos = cur;
113 break;
114 }
115 prev = cur;
116 }
117
118 blk_mq_debugfs_unregister_rqos(rqos);
119}

170static inline void rq_qos_throttle(struct request_queue *q, struct bio *bio)
171{
172 /*
173 * BIO_TRACKED lets controllers know that a bio went through the
174 * normal rq_qos path.
175 */
176 bio_set_flag(bio, BIO_TRACKED);
177 if (q->rq_qos)
178 __rq_qos_throttle(q->rq_qos, bio);
179}
