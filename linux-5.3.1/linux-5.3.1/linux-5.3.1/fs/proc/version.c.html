2#include <linux/fs.h>

9static int version_proc_show(struct seq_file *m, void *v)
10{
11 seq_printf(m, linux_proc_banner,
12 utsname()->sysname,
13 utsname()->release,
14 utsname()->version);
15 return 0;
16}
17
18static int __init proc_version_init(void)
19{
20 proc_create_single("version", 0, NULL, version_proc_show);
21 return 0;
22}
23fs_initcall(proc_version_init);
