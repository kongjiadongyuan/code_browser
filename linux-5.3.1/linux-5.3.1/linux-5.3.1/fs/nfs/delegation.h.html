81static inline int nfs_have_delegated_attributes(struct inode *inode)
82{
83 return NFS_PROTO(inode)->have_delegation(inode, FMODE_READ) &&
84 !(NFS_I(inode)->cache_validity & NFS_INO_REVAL_FORCED);
85}
