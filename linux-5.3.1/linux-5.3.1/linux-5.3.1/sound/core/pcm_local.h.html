38static inline snd_pcm_uframes_t
39snd_pcm_avail(struct snd_pcm_substream *substream)
40{
41 if (substream->stream == SNDRV_PCM_STREAM_PLAYBACK)
42 return snd_pcm_playback_avail(substream->runtime);
43 else
44 return snd_pcm_capture_avail(substream->runtime);
45}

47static inline snd_pcm_uframes_t
48snd_pcm_hw_avail(struct snd_pcm_substream *substream)
49{
50 if (substream->stream == SNDRV_PCM_STREAM_PLAYBACK)
51 return snd_pcm_playback_hw_avail(substream->runtime);
52 else
53 return snd_pcm_capture_hw_avail(substream->runtime);
54}
