44static __always_inline void clac(void)
45{
46 /* Note: a barrier is implicit in alternative() */
47 alternative("", __ASM_CLAC, X86_FEATURE_SMAP);
48}
49
50static __always_inline void stac(void)
51{
52 /* Note: a barrier is implicit in alternative() */
53 alternative("", __ASM_STAC, X86_FEATURE_SMAP);
54}
55
56static __always_inline unsigned long smap_save(void)
57{
58 unsigned long flags;
59
60 asm volatile (ALTERNATIVE("", "pushf; pop %0; " __ASM_CLAC,
61 X86_FEATURE_SMAP)
62 : "=rm" (flags) : : "memory", "cc");
63
64 return flags;
65}
66
67static __always_inline void smap_restore(unsigned long flags)
68{
69 asm volatile (ALTERNATIVE("", "push %0; popf", X86_FEATURE_SMAP)
70 : : "g" (flags) : "memory", "cc");
71}
72
73/* These macros can be used in asm() statements */
74#define ASM_CLAC \
75 ALTERNATIVE("", __ASM_CLAC, X86_FEATURE_SMAP)
76#define ASM_STAC \
77 ALTERNATIVE("", __ASM_STAC, X86_FEATURE_SMAP)
78
79#else /* CONFIG_X86_SMAP */
80
81static inline void clac(void) { }
82static inline void stac(void) { }
83
84static inline unsigned long smap_save(void) { return 0; }
85static inline void smap_restore(unsigned long flags) { }
