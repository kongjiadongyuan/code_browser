31static int __init early_initrd(char *p)
32{
33 phys_addr_t start;
34 unsigned long size;
35 char *endp;
36
37 start = memparse(p, &endp);
38 if (*endp == ',') {
39 size = memparse(endp + 1, NULL);
40
41 phys_initrd_start = start;
42 phys_initrd_size = size;
43 }
44 return 0;
45}
46early_param("initrd", early_initrd);

48static int init_linuxrc(struct subprocess_info *info, struct cred *new)
49{
50 ksys_unshare(CLONE_FS | CLONE_FILES);
51 /* stdin/stdout/stderr for /linuxrc */
52 ksys_open("/dev/console", O_RDWR, 0);
53 ksys_dup(0);
54 ksys_dup(0);
55 /* move initrd over / and chdir/chroot in initrd root */
56 ksys_chdir("/root");
57 ksys_mount(".", "/", NULL, MS_MOVE, NULL);
58 ksys_chroot(".");
59 ksys_setsid();
60 return 0;
61}

63static void __init handle_initrd(void)
64{
65 struct subprocess_info *info;
66 static char *argv[] = { "linuxrc", NULL, };
67 extern char *envp_init[];
68 int error;
69
70 real_root_dev = new_encode_dev(ROOT_DEV);
71 create_dev("/dev/root.old", Root_RAM0);
72 /* mount initrd on rootfs' /root */
73 mount_block_root("/dev/root.old", root_mountflags & ~MS_RDONLY);
74 ksys_mkdir("/old", 0700);
75 ksys_chdir("/old");
76
77 /*
78 * In case that a resume from disk is carried out by linuxrc or one of
79 * its children, we need to tell the freezer not to wait for us.
80 */
81 current->flags |= PF_FREEZER_SKIP;
82
83 info = call_usermodehelper_setup("/linuxrc", argv, envp_init,
84 GFP_KERNEL, init_linuxrc, NULL, NULL);
85 if (!info)
86 return;
87 call_usermodehelper_exec(info, UMH_WAIT_PROC);
88
89 current->flags &= ~PF_FREEZER_SKIP;
90
91 /* move initrd to rootfs' /old */
92 ksys_mount("..", ".", NULL, MS_MOVE, NULL);
93 /* switch root and cwd back to / of rootfs */
94 ksys_chroot("..");
95
96 if (new_decode_dev(real_root_dev) == Root_RAM0) {
97 ksys_chdir("/old");
98 return;
99 }
100
101 ksys_chdir("/");
102 ROOT_DEV = new_decode_dev(real_root_dev);
103 mount_root();
104
105 printk(KERN_NOTICE "Trying to move old root to /initrd ... ");
106 error = ksys_mount("/old", "/root/initrd", NULL, MS_MOVE, NULL);
107 if (!error)
108 printk("okay\n");
109 else {
110 int fd = ksys_open("/dev/root.old", O_RDWR, 0);
111 if (error == -ENOENT)
112 printk("/initrd does not exist. Ignored.\n");
113 else
114 printk("failed\n");
115 printk(KERN_NOTICE "Unmounting old root\n");
116 ksys_umount("/old", MNT_DETACH);
117 printk(KERN_NOTICE "Trying to free ramdisk memory ... ");
118 if (fd < 0) {
119 error = fd;
120 } else {
121 error = ksys_ioctl(fd, BLKFLSBUF, 0);
122 ksys_close(fd);
123 }
124 printk(!error ? "okay\n" : "failed\n");
125 }
126}

128bool __init initrd_load(void)
129{
130 if (mount_initrd) {
131 create_dev("/dev/ram", Root_RAM0);
132 /*
133 * Load the initrd data into /dev/ram0. Execute it as initrd
134 * unless /dev/ram0 is supposed to be our actual root device,
135 * in that case the ram disk is just set up here, and gets
136 * mounted in the normal path.
137 */
138 if (rd_load_image("/initrd.image") && ROOT_DEV != Root_RAM0) {
139 ksys_unlink("/initrd.image");
140 handle_initrd();
141 return true;
142 }
143 }
144 ksys_unlink("/initrd.image");
145 return false;
146}
