34/*
35 * codec bus
36 *
37 * each controller needs to creata a hda_bus to assign the accessor.
38 * A hda_bus contains several codecs in the list codec_list.
39 */
40struct hda_bus {
41 struct hdac_bus core;
42
43 struct snd_card *card;
44
45 struct pci_dev *pci;
46 const char *modelname;
47
48 struct mutex prepare_mutex;
49
50 /* assigned PCMs */
51 DECLARE_BITMAP(pcm_dev_bits, SNDRV_PCM_DEVICES);
52
53 /* misc op flags */
54 unsigned int needs_damn_long_delay :1;
55 unsigned int allow_bus_reset:1; /* allow bus reset at fatal error */
56 /* status for codec/controller */
57 unsigned int shutdown :1; /* being unloaded */
58 unsigned int response_reset:1; /* controller was reset */
59 unsigned int in_reset:1; /* during reset operation */
60 unsigned int no_response_fallback:1; /* don't fallback at RIRB error */
61 unsigned int bus_probing :1; /* during probing process */
62
63 int primary_dig_out_type; /* primary digital out PCM type */
64 unsigned int mixer_assigned; /* codec addr for mixer name */
65};

137/* PCM information for each substream */
138struct hda_pcm_stream {
139 unsigned int substreams; /* number of substreams, 0 = not exist*/
140 unsigned int channels_min; /* min. number of channels */
141 unsigned int channels_max; /* max. number of channels */
142 hda_nid_t nid; /* default NID to query rates/formats/bps, or set up */
143 u32 rates; /* supported rates */
144 u64 formats; /* supported formats (SNDRV_PCM_FMTBIT_) */
145 unsigned int maxbps; /* supported max. bit per sample */
146 const struct snd_pcm_chmap_elem *chmap; /* chmap to override */
147 struct hda_pcm_ops ops;
148};
