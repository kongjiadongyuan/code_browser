152/**
153 * drm_atomic_crtc_for_each_plane - iterate over planes currently attached to CRTC
154 * @plane: the loop cursor
155 * @crtc: the crtc whose planes are iterated
156 *
157 * This iterates over the current state, useful (for example) when applying
158 * atomic state after it has been checked and swapped. To iterate over the
159 * planes which *will* be attached (more useful in code called from
160 * &drm_mode_config_funcs.atomic_check) see
161 * drm_atomic_crtc_state_for_each_plane().
162 */
163#define drm_atomic_crtc_for_each_plane(plane, crtc) \
164 drm_for_each_plane_mask(plane, (crtc)->dev, (crtc)->state->plane_mask)
165
166/**
167 * drm_crtc_atomic_state_for_each_plane - iterate over attached planes in new state
168 * @plane: the loop cursor
169 * @crtc_state: the incoming crtc-state
170 *
171 * Similar to drm_crtc_for_each_plane(), but iterates the planes that will be
172 * attached if the specified state is applied. Useful during for example
173 * in code called from &drm_mode_config_funcs.atomic_check operations, to
174 * validate the incoming state.
175 */
176#define drm_atomic_crtc_state_for_each_plane(plane, crtc_state) \
177 drm_for_each_plane_mask(plane, (crtc_state)->state->dev, (crtc_state)->plane_mask)
178
179/**
180 * drm_crtc_atomic_state_for_each_plane_state - iterate over attached planes in new state
181 * @plane: the loop cursor
182 * @plane_state: loop cursor for the plane's state, must be const
183 * @crtc_state: the incoming crtc-state
184 *
185 * Similar to drm_crtc_for_each_plane(), but iterates the planes that will be
186 * attached if the specified state is applied. Useful during for example
187 * in code called from &drm_mode_config_funcs.atomic_check operations, to
188 * validate the incoming state.
189 *
190 * Compared to just drm_atomic_crtc_state_for_each_plane() this also fills in a
191 * const plane_state. This is useful when a driver just wants to peek at other
192 * active planes on this crtc, but does not need to change it.
193 */
194#define drm_atomic_crtc_state_for_each_plane_state(plane, plane_state, crtc_state) \
195 drm_for_each_plane_mask(plane, (crtc_state)->state->dev, (crtc_state)->plane_mask) \
196 for_each_if ((plane_state = \
197 __drm_atomic_get_current_plane_state((crtc_state)->state, \
198 plane)))

200/**
201 * drm_atomic_plane_disabling - check whether a plane is being disabled
202 * @old_plane_state: old atomic plane state
203 * @new_plane_state: new atomic plane state
204 *
205 * Checks the atomic state of a plane to determine whether it's being disabled
206 * or not. This also WARNs if it detects an invalid state (both CRTC and FB
207 * need to either both be NULL or both be non-NULL).
208 *
209 * RETURNS:
210 * True if the plane is being disabled, false otherwise.
211 */
212static inline bool
213drm_atomic_plane_disabling(struct drm_plane_state *old_plane_state,
214 struct drm_plane_state *new_plane_state)
215{
216 /*
217 * When disabling a plane, CRTC and FB should always be NULL together.
218 * Anything else should be considered a bug in the atomic core, so we
219 * gently warn about it.
220 */
221 WARN_ON((new_plane_state->crtc == NULL && new_plane_state->fb != NULL) ||
222 (new_plane_state->crtc != NULL && new_plane_state->fb == NULL));
223
224 return old_plane_state->crtc && !new_plane_state->crtc;
225}
