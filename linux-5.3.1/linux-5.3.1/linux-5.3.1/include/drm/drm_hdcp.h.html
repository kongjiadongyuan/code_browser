250/*
251 * Helper functions to convert 24bit big endian hdcp sequence number to
252 * host format and back
253 */
254static inline
255u32 drm_hdcp_be24_to_cpu(const u8 seq_num[HDCP_2_2_SEQ_NUM_LEN])
256{
257 return (u32)(seq_num[2] | seq_num[1] << 8 | seq_num[0] << 16);
258}
259
260static inline
261void drm_hdcp_cpu_to_be24(u8 seq_num[HDCP_2_2_SEQ_NUM_LEN], u32 val)
262{
263 seq_num[0] = val >> 16;
264 seq_num[1] = val >> 8;
265 seq_num[2] = val;
266}
