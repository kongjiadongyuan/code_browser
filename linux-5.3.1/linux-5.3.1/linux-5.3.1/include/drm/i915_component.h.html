24#ifndef _I915_COMPONENT_H_
25#define _I915_COMPONENT_H_
26
27#include "drm_audio_component.h"
28
29enum i915_component_type {
30 I915_COMPONENT_AUDIO = 1,
31 I915_COMPONENT_HDCP,
32};
33
34/* MAX_PORT is the number of port
35 * It must be sync with I915_MAX_PORTS defined i915_drv.h
36 */
37#define MAX_PORTS 6
38
39/**
40 * struct i915_audio_component - Used for direct communication between i915 and hda drivers
41 */
42struct i915_audio_component {
43 /**
44 * @base: the drm_audio_component base class
45 */
46 struct drm_audio_component base;
47
48 /**
49 * @aud_sample_rate: the array of audio sample rate per port
50 */
51 int aud_sample_rate[MAX_PORTS];
52};
53
54#endif /* _I915_COMPONENT_H_ */
