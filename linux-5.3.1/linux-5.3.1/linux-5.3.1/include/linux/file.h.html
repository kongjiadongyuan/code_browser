34struct fd {
35 struct file *file;
36 unsigned int flags;
37};
38#define FDPUT_FPUT 1
39#define FDPUT_POS_UNLOCK 2
40
41static inline void fdput(struct fd fd)
42{
43 if (fd.flags & FDPUT_FPUT)
44 fput(fd.file);
45}

55static inline struct fd __to_fd(unsigned long v)
56{
57 return (struct fd){(struct file *)(v & ~3),v & 3};
58}
59
60static inline struct fd fdget(unsigned int fd)
61{
62 return __to_fd(__fdget(fd));
63}
64
65static inline struct fd fdget_raw(unsigned int fd)
66{
67 return __to_fd(__fdget_raw(fd));
68}
69
70static inline struct fd fdget_pos(int fd)
71{
72 return __to_fd(__fdget_pos(fd));
73}
74
75static inline void fdput_pos(struct fd f)
76{
77 if (f.flags & FDPUT_POS_UNLOCK)
78 __f_unlock_pos(f.file);
79 fdput(f);
80}
