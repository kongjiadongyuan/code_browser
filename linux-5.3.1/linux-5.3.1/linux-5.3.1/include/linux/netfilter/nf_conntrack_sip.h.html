74#define __SIP_HDR(__name, __cname, __search, __match) \
75{ \
76 .name = (__name), \

79 .clen = (__cname) ? sizeof(__cname) - 1 : 0, \
80 .search = (__search), \
81 .slen = (__search) ? sizeof(__search) - 1 : 0, \
82 .match_len = (__match), \
83}
