62static inline bool daxdev_mapping_supported(struct vm_area_struct *vma,
63 struct dax_device *dax_dev)
64{
65 if (!(vma->vm_flags & VM_SYNC))
66 return true;
67 if (!IS_DAX(file_inode(vma->vm_file)))
68 return false;
69 return dax_synchronous(dax_dev);
70}
