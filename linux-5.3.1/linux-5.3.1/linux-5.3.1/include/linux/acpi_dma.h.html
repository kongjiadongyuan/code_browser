19/**
20 * struct acpi_dma_spec - slave device DMA resources
21 * @chan_id: channel unique id
22 * @slave_id: request line unique id
23 * @dev: struct device of the DMA controller to be used in the filter
24 * function
25 */
26struct acpi_dma_spec {
27 int chan_id;
28 int slave_id;
29 struct device *dev;
30};
31
32/**
33 * struct acpi_dma - representation of the registered DMAC
34 * @dma_controllers: linked list node
35 * @dev: struct device of this controller
36 * @acpi_dma_xlate: callback function to find a suitable channel
37 * @data: private data used by a callback function
38 * @base_request_line: first supported request line (CSRT)
39 * @end_request_line: last supported request line (CSRT)
40 */
41struct acpi_dma {
42 struct list_head dma_controllers;
43 struct device *dev;
44 struct dma_chan *(*acpi_dma_xlate)
45 (struct acpi_dma_spec *, struct acpi_dma *);
46 void *data;
47 unsigned short base_request_line;
48 unsigned short end_request_line;
49};
50
51/* Used with acpi_dma_simple_xlate() */
52struct acpi_dma_filter_info {
53 dma_cap_mask_t dma_cap;
54 dma_filter_fn filter_fn;
55};
