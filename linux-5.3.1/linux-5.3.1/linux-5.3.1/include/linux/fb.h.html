218/*
219 * Frame buffer operations
220 *
221 * LOCKING NOTE: those functions must _ALL_ be called with the console
222 * semaphore held, this is the only suitable locking mechanism we have
223 * in 2.6. Some may be called at interrupt time at this point though.
224 *
225 * The exception to this is the debug related hooks. Putting the fb
226 * into a debug state (e.g. flipping to the kernel console) and restoring
227 * it must be done in a lock-free manner, so low level drivers should
228 * keep track of the initial console (if applicable) and may need to
229 * perform direct, unlocked hardware writes in these hooks.
230 */
231

239 * work with normal memory mapped access
240 */

242 size_t count, loff_t *ppos);
243 ssize_t (*fb_write)(struct fb_info *info, const char __user *buf,
244 size_t count, loff_t *ppos);
245
246 /* checks var and eventually tweaks it to something supported,
247 * DO NOT MODIFY PAR */
248 int (*fb_check_var)(struct fb_var_screeninfo *var, struct fb_info *info);
249
250 /* set the video mode according to info->var */

379/* FBINFO_* = fb_info.flags bit flags */
380#define FBINFO_DEFAULT 0
381#define FBINFO_HWACCEL_DISABLED 0x0002
382 /* When FBINFO_HWACCEL_DISABLED is set:
383 * Hardware acceleration is turned off. Software implementations
384 * of required functions (copyarea(), fillrect(), and imageblit())
385 * takes over; acceleration engine should be in a quiescent state */
386
387/* hints */
388#define FBINFO_VIRTFB 0x0004 /* FB is System RAM, not device. */
389#define FBINFO_PARTIAL_PAN_OK 0x0040 /* otw use pan only for double-buffering */
390#define FBINFO_READS_FAST 0x0080 /* soft-copy faster than rendering */
391
392/* hardware supported ops */
393/* semantics: when a bit is set, it indicates that the operation is
394 * accelerated by hardware.
395 * required functions will still work even if the bit is not set.
396 * optional functions may not even exist if the flag bit is not set.
397 */
398#define FBINFO_HWACCEL_NONE 0x0000
399#define FBINFO_HWACCEL_COPYAREA 0x0100 /* required */
400#define FBINFO_HWACCEL_FILLRECT 0x0200 /* required */
401#define FBINFO_HWACCEL_IMAGEBLIT 0x0400 /* required */
402#define FBINFO_HWACCEL_ROTATE 0x0800 /* optional */
403#define FBINFO_HWACCEL_XPAN 0x1000 /* optional */
404#define FBINFO_HWACCEL_YPAN 0x2000 /* optional */
405#define FBINFO_HWACCEL_YWRAP 0x4000 /* optional */
406
407#define FBINFO_MISC_USEREVENT 0x10000 /* event request
408 from userspace */
409#define FBINFO_MISC_TILEBLITTING 0x20000 /* use tile blitting */
410
411/* A driver may set this flag to indicate that it does want a set_par to be
412 * called every time when fbcon_switch is executed. The advantage is that with
413 * this flag set you can really be sure that set_par is always called before
414 * any of the functions dependent on the correct hardware state or altering
415 * that state, even if you are using some broken X releases. The disadvantage
416 * is that it introduces unwanted delays to every console switch if set_par
417 * is slow. It is a good idea to try this flag in the drivers initialization
418 * code whenever there is a bug report related to switching between X and the
419 * framebuffer console.
420 */
421#define FBINFO_MISC_ALWAYS_SETPAR 0x40000
422
423/* where the fb is a firmware driver, and can be replaced with a proper one */
424#define FBINFO_MISC_FIRMWARE 0x80000
425/*
426 * Host and GPU endianness differ.
427 */
428#define FBINFO_FOREIGN_ENDIAN 0x100000
429/*
430 * Big endian math. This is the same flags as above, but with different
431 * meaning, it is set by the fb subsystem depending FOREIGN_ENDIAN flag
432 * and host endianness. Drivers should not use this flag.
433 */
434#define FBINFO_BE_MATH 0x100000
435/*
436 * Hide smem_start in the FBIOGET_FSCREENINFO IOCTL. This is used by modern DRM
437 * drivers to stop userspace from trying to share buffers behind the kernel's
438 * back. Instead dma-buf based buffer sharing should be used.
439 */
440#define FBINFO_HIDE_SMEM_START 0x200000

649static inline void __fb_pad_aligned_buffer(u8 *dst, u32 d_pitch,
650 u8 *src, u32 s_pitch, u32 height)
651{
652 u32 i, j;
653
654 d_pitch -= s_pitch;
655
656 for (i = height; i--; ) {
657 /* s_pitch is a few bytes at the most, memcpy is suboptimal */
658 for (j = 0; j < s_pitch; j++)
659 *dst++ = *src++;
660 dst += d_pitch;
661 }
662}
