152/*
153 * Protects ->fs, ->files, ->mm, ->group_info, ->comm, keyring
154 * subscriptions and synchronises with wait4(). Also used in procfs. Also
155 * pins the final release of task.io_context. Also protects ->cpuset and
156 * ->cgroup.subsys[]. And ->vfork_done.
157 *
158 * Nests both inside and outside of read_lock(&tasklist_lock).
159 * It must not be nested with write_lock_irq(&tasklist_lock),
160 * neither inside nor outside.
161 */
162static inline void task_lock(struct task_struct *p)
163{
164 spin_lock(&p->alloc_lock);
165}
166
167static inline void task_unlock(struct task_struct *p)
168{
169 spin_unlock(&p->alloc_lock);
170}
