11#include <uapi/linux/uuid.h>

14#define UUID_SIZE 16
15
16typedef struct {
17 __u8 b[UUID_SIZE];
18} uuid_t;
19
20#define UUID_INIT(a, b, c, d0, d1, d2, d3, d4, d5, d6, d7) \
21((uuid_t) \
22{{ ((a) >> 24) & 0xff, ((a) >> 16) & 0xff, ((a) >> 8) & 0xff, (a) & 0xff, \
23 ((b) >> 8) & 0xff, (b) & 0xff, \
24 ((c) >> 8) & 0xff, (c) & 0xff, \
25 (d0), (d1), (d2), (d3), (d4), (d5), (d6), (d7) }})
26
27/*
28 * The length of a UUID string ("aaaaaaaa-bbbb-cccc-dddd-eeeeeeeeeeee")
29 * not including trailing NUL.
30 */
31#define UUID_STRING_LEN 36
32
33extern const guid_t guid_null;
34extern const uuid_t uuid_null;
35
36static inline bool guid_equal(const guid_t *u1, const guid_t *u2)
37{
38 return memcmp(u1, u2, sizeof(guid_t)) == 0;
39}
40
41static inline void guid_copy(guid_t *dst, const guid_t *src)
42{
43 memcpy(dst, src, sizeof(guid_t));
44}
45
46static inline bool guid_is_null(const guid_t *guid)
47{
48 return guid_equal(guid, &guid_null);
49}
50
51static inline bool uuid_equal(const uuid_t *u1, const uuid_t *u2)
52{
53 return memcmp(u1, u2, sizeof(uuid_t)) == 0;
54}
55
56static inline void uuid_copy(uuid_t *dst, const uuid_t *src)
57{
58 memcpy(dst, src, sizeof(uuid_t));
59}
60
61static inline bool uuid_is_null(const uuid_t *uuid)
62{
63 return uuid_equal(uuid, &uuid_null);
64}

66void generate_random_uuid(unsigned char uuid[16]);

69extern void uuid_gen(uuid_t *u);

71bool __must_check uuid_is_valid(const char *uuid);

74extern const u8 uuid_index[16];
