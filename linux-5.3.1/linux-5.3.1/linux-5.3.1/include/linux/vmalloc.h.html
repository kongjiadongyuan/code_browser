16/* bits in flags of vmalloc's vm_struct below */
17#define VM_IOREMAP 0x00000001 /* ioremap() and friends */
18#define VM_ALLOC 0x00000002 /* vmalloc() */
19#define VM_MAP 0x00000004 /* vmap()ed pages */
20#define VM_USERMAP 0x00000008 /* suitable for remap_vmalloc_range */
21#define VM_UNINITIALIZED 0x00000020 /* vm_struct is not fully initialized */
22#define VM_NO_GUARD 0x00000040 /* don't add guard page */
23#define VM_KASAN 0x00000080 /* has allocated kasan shadow memory */
24/*
25 * Memory with VM_FLUSH_RESET_PERMS cannot be freed in an interrupt or with
26 * vfree_atomic().
27 */
28#define VM_FLUSH_RESET_PERMS 0x00000100 /* Reset direct map and flush TLB on unmap */
29/* bits [20..32] reserved for arch specific ioremap internals */
30
31/*
32 * Maximum alignment for ioremap() regions.
33 * Can be overriden by arch-specific value.
34 */
35#ifndef IOREMAP_MAX_ORDER
36#define IOREMAP_MAX_ORDER (7 + PAGE_SHIFT) /* 128 pages */
37#endif

39struct vm_struct {
40 struct vm_struct *next;
41 void *addr;
42 unsigned long size;
43 unsigned long flags;
44 struct page **pages;
45 unsigned int nr_pages;
46 phys_addr_t phys_addr;
47 const void *caller;
48};

50struct vmap_area {
51 unsigned long va_start;
52 unsigned long va_end;
53
54 /*
55 * Largest available free size in subtree.
56 */
57 unsigned long subtree_max_size;
58 unsigned long flags;

63};

127static inline size_t get_vm_area_size(const struct vm_struct *area)
128{
129 if (!(area->flags & VM_NO_GUARD))
130 /* return actual size without guard page */
131 return area->size - PAGE_SIZE;
132 else
133 return area->size;
134
135}
