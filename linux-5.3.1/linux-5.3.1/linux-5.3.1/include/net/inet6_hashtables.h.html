27static inline unsigned int __inet6_ehashfn(const u32 lhash,
28 const u16 lport,
29 const u32 fhash,
30 const __be16 fport,
31 const u32 initval)
32{
33 const u32 ports = (((u32)lport) << 16) | (__force u32)fport;
34 return jhash_3words(lhash, fhash, ports, initval);
35}

60static inline struct sock *__inet6_lookup(struct net *net,
61 struct inet_hashinfo *hashinfo,
62 struct sk_buff *skb, int doff,
63 const struct in6_addr *saddr,
64 const __be16 sport,
65 const struct in6_addr *daddr,
66 const u16 hnum,
67 const int dif, const int sdif,
68 bool *refcounted)
69{
70 struct sock *sk = __inet6_lookup_established(net, hashinfo, saddr,
71 sport, daddr, hnum,
72 dif, sdif);
73 *refcounted = true;
74 if (sk)
75 return sk;
76 *refcounted = false;
77 return inet6_lookup_listener(net, hashinfo, skb, doff, saddr, sport,
78 daddr, hnum, dif, sdif);
79}
80
81static inline struct sock *__inet6_lookup_skb(struct inet_hashinfo *hashinfo,
82 struct sk_buff *skb, int doff,
83 const __be16 sport,
84 const __be16 dport,
85 int iif, int sdif,
86 bool *refcounted)
87{
88 struct sock *sk = skb_steal_sock(skb);
89
90 *refcounted = true;
91 if (sk)
92 return sk;
93
94 return __inet6_lookup(dev_net(skb_dst(skb)->dev), hashinfo, skb,
95 doff, &ipv6_hdr(skb)->saddr, sport,
96 &ipv6_hdr(skb)->daddr, ntohs(dport),
97 iif, sdif, refcounted);
98}

109#define INET6_MATCH(__sk, __net, __saddr, __daddr, __ports, __dif, __sdif) \
110 (((__sk)->sk_portpair == (__ports)) && \
111 ((__sk)->sk_family == AF_INET6) && \
112 ipv6_addr_equal(&(__sk)->sk_v6_daddr, (__saddr)) && \
113 ipv6_addr_equal(&(__sk)->sk_v6_rcv_saddr, (__daddr)) && \
114 (((__sk)->sk_bound_dev_if == (__dif)) || \
115 ((__sk)->sk_bound_dev_if == (__sdif))) && \
116 net_eq(sock_net(__sk), (__net)))
